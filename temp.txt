bool hitboxColide(GameObject object1, GameObject object2) {
    // Funkcja pomocnicza do pobierania punktu w przestrzeni świata
    auto getAbs = [](GameObject& obj, PixelVector corner) {
        return Point{ obj.sprite.x + corner.sX, obj.sprite.y + corner.sY };
    };

    Hitbox h1 = object1.getCurrentCostume().hitbox;
    Hitbox h2 = object2.getCurrentCostume().hitbox;

    // Punkty obu hitboxów
    Point pts1[4] = { getAbs(object1, h1.topLeft), getAbs(object1, h1.topRight), 
                      getAbs(object1, h1.bottomRight), getAbs(object1, h1.bottomLeft) };
    Point pts2[4] = { getAbs(object2, h2.topLeft), getAbs(object2, h2.topRight), 
                      getAbs(object2, h2.bottomRight), getAbs(object2, h2.bottomLeft) };

    // Tablica punktów do łatwiejszej iteracji
    Point* shapes[2] = { pts1, pts2 };

    for (int s = 0; s < 2; s++) {
        Point* currentShape = shapes[s];
        
        // Sprawdzamy 2 osie dla każdego prostokąta (szerokość i wysokość)
        for (int i = 0; i < 2; i++) {
            Point p1 = currentShape[i];
            Point p2 = currentShape[i + 1];

            // Wyznaczamy oś prostopadłą do krawędzi
            int axisX = -(p2.y - p1.y);
            int axisY = (p2.x - p1.x);

            // Rzutowanie (Iloczyn skalarny)
            auto project = [&](Point* pts, int ax, int ay) {
                int min = pts[0].x * ax + pts[0].y * ay;
                int max = min;
                for (int j = 1; j < 4; j++) {
                    int proj = pts[j].x * ax + pts[j].y * ay;
                    if (proj < min) min = proj;
                    if (proj > max) max = proj;
                }
                return std::make_pair(min, max);
            };

            auto res1 = project(pts1, axisX, axisY);
            auto res2 = project(pts2, axisX, axisY);

            // Test separacji
            if (res1.second < res2.first || res2.second < res1.first) {
                return false; // Znaleziono przerwę -> brak kolizji
            }
        }
    }

    return true; // Brak przerwy na wszystkich osiach -> kolizja
}
